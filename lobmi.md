i wrote a script to reverse this

```
import wave

song = wave.open('lobmi.wav', mode='rb')
frame_bytes = bytearray(list(song.readframes(song.getnframes())))

bits = ""
print(frame_bytes)
for byte in frame_bytes:
    print(byte)
    bits += bin(byte)[-1]  # Extracting the least significant bit from each byte

# Retrieve the hidden message length
msg_length = int(bits[:64], 2) * 8

# Retrieve the hidden message bits
hidden_bits = bits[64:64 + msg_length]

# Convert the bit sequence back to characters
hidden_text = ''.join(chr(int(hidden_bits[i:i+8], 2)) for i in range(0, len(hidden_bits), 8))

print("Hidden text:", hidden_text)

# Extract the original audio data by resetting the LSBs to zero
for i in range(len(hidden_bits)):
    index = i + 64
    if index < len(frame_bytes) * 8:
        bit = int(hidden_bits[i])
        frame_bytes[index // 8] = frame_bytes[index // 8] & ~1 | bit

frame_modified = bytes(frame_bytes)
with wave.open('restored_orig.wav', 'wb') as fd:
    fd.setparams(song.getparams())
    fd.writeframes(frame_modified)

song.close()
```
